import "react-native-gesture-handler";
import { View, Text, Image } from "react-native";
import {
  SimpleLineIcons,
  MaterialIcons,
  MaterialCommunityIcons,
  FontAwesome
} from "@expo/vector-icons";
import { SafeAreaView } from "react-native-safe-area-context";
import { NavigationContainer } from "@react-navigation/native";
import { createDrawerNavigator, DrawerContentScrollView, DrawerItem } from "@react-navigation/drawer";
import User from "./assets/user.jpg";
import Home from "./screens/Home";
import Settings from "./screens/Settings";
import LoginUser from "./screens/LoginUser";
import OrderDetails from "./screens/OrderDetails"

const Drawer = createDrawerNavigator();

// Defina um conjunto de rotas que deseja exibir no Drawer Navigator teste
const routesToShowInDrawer = ["Home", "Settings"];

const CustomDrawerContent = (props) => {
  const filteredRoutes = props.state.routes.filter(route =>
    routesToShowInDrawer.includes(route.name)
  );

  return (
    <DrawerContentScrollView {...props}>
      <SafeAreaView>
        <View
          style={{
            height: 200,
            width: '100%',
            justifyContent: "center",
            alignItems: "center",
            borderBottomColor: "#f4f4f4",
            borderBottomWidth: 1
          }}
        >
          <Image
            source={User}
            style={{
              height: 130,
              width: 130,
              borderRadius: 65
            }}
          />
          <Text
            style={{
              fontSize: 22,
              marginVertical: 6,
              fontWeight: "bold",
              color: "#111"
            }}
          >Isabella Joanna</Text>
          <Text
            style={{
              fontSize: 16,
              color: "#111"
            }}
          >Técnico 1</Text>
        </View>
        
        {filteredRoutes.map((route, index) => (
          <DrawerItem
            label={route.drawerIcon + route.name}
            key={index}
            onPress={() => {
              props.navigation.navigate(route.name);
            }}
          />
        ))}
      </SafeAreaView>
    </DrawerContentScrollView>
  );
}

export default function App() {
  return (
    <NavigationContainer>
      <Drawer.Navigator
        initialRouteName="LoginUser"
        drawerContent={props => <CustomDrawerContent {...props} />}
        drawerStyle={{
          backgroundColor: "#fff",
          width: 250
        }}
        screenOptions={{
          headerStyle: {
            backgroundColor: "#4E54C8",
          },
          headerTintColor: "#fff",
          headerTitleStyle: {
            fontWeight: "bold"
          },
          drawerLabelStyle: {
            color: "#111"
          }
        }}
      >
        <Drawer.Screen
          name="Home"
          options={{
            drawerLabel: "Home",
            title: "SOS DOS ELEVADORES",
            drawerIcon: () => (
              <SimpleLineIcons name="home" size={20} color="#808080" />
            )
          }}
          component={Home}
        />

        <Drawer.Screen
          name="Settings"
          options={{
            drawerLabel: "Settings",
            title: "Settings",
            drawerIcon: () => (
              <SimpleLineIcons name="settings" size={20} color="#808080" />
            )
          }}
          component={Settings}
        />

        <Drawer.Screen
          name="LoginUser"
          options={{
            headerShown: false,
            drawerLabel: "login",
            title: "login",
            drawerIcon: () => (
              <SimpleLineIcons name="settings" size={20} color="#808080" />
            )
          }}
          component={LoginUser}
        />

        <Drawer.Screen
          name="OrderDetails"
          options={{
            headerShown: true,
            drawerLabel: "OrderDetails",
            title: "OrderDetails",
            drawerIcon: () => (
              <SimpleLineIcons name="home" size={20} color="#808080" />
            )
          }}
          component={OrderDetails}
        />
      </Drawer.Navigator>
    </NavigationContainer>
  );
}
